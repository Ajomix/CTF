- Bài này tương tự thuật toán [TEA](https://en.wikipedia.org/wiki/Tiny_Encryption_Algorithm) nhưng thay đổi các hằng số 

```C
#include <stdio.h>
#include <string.h>

unsigned char buf[] =
{
  0x10, 0xA3, 0x5C, 0x57, 0x0D, 0x35, 0xD5, 0x49, 0x6A, 0x0F,
  0xF9, 0x9E, 0x8E, 0x20, 0xFC, 0xBF, 0xA2, 0xB7, 0x0C, 0xBB,
  0xC7, 0x81, 0x46, 0x64, 0xEA, 0x1B, 0xB6, 0xE0, 0x97, 0x38,
  0xE8, 0x83, 0xDD, 0x6D, 0x9F, 0x5A, 0x5E, 0x36, 0xCE, 0x13,
  0x3D, 0xF4, 0xD0, 0x16, 0x2B, 0x5F, 0x57, 0x5B, 0xC0, 0x57,
  0x15, 0xE4, 0xC7, 0xE3, 0x37, 0xE5, 0xDE, 0x30, 0xC1, 0x71,
  0x25, 0xA5, 0xBD, 0x89, 0x5F, 0xE1, 0x19, 0x49, 0x6C, 0x3C,
  0x9B, 0x2C, 0x51, 0xE8, 0x77, 0xD7, 0x11, 0x94, 0xA0, 0x30
};
int main()
{
	unsigned long long buf_1;
	unsigned long long buf_0;
	long long v13=0;
	

	for (int i = 0; i < 32; i++) {
		v13 += 0x123457898765432;
	}
	long long sample = v13;
	long long index = 4;

	for (int i = 0; i < 5; i++) {
		buf_0 = *((long long*)(buf + index * 16));
		buf_1 = *((long long*)(buf + index * 16) + 1);
		
		v13 = sample;
		for (int j = 0; j < 32; j++) {
			buf_1 -= (v13 + buf_0) ^ (16 * buf_0 - 0x2CBDA60BFD707FD3) ^ ((buf_0 >> 5) + 0x424F0D99A012A826);
			buf_0 -= (v13 + buf_1) ^ (16 * buf_1 - 0x1EC32D622D0480D8) ^ ((buf_1 >> 5) + 0x488D27F32AE91451);
			v13 -= 0x123457898765432;
		}
		*((long long*)(buf + index * 16)) = buf_0;
		*((long long*)(buf + index * 16) + 1) = buf_1;
		index--;
	}
	
	printf("%s", buf);
	
	return 0;
}
```
